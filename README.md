## Install:
```bash
npm i gradient-background
---
yarn add gradient-background
```

## Usage:
```jsx
// don't forget change props
import { GradientBackgroud } from 'gradient-background';


const App = () => {
  return (
    <div>
      <GradientBackgroud color='primary'>
    </div>
  );
};
```
